  <script>
    const form = document.getElementById("quiz");
    const result = document.getElementById("result");
    const scoreEl = document.getElementById("score");
    const messageEl = document.getElementById("message");

    form.addEventListener("submit", e => {
      e.preventDefault();

      // 適当にスコア算出（ランダム補正付き）
      let base = 0;
      const data = new FormData(form);

      if (data.get("height") == "177.8" || data.get("height") == "177.6") base += 25;
      if (data.get("blood") == "A") base += 25;
      if (data.get("car") == "アウディ") base += 25;
      if (data.get("soccer") == "新宿の心臓" || data.get("soccer") == "鉄壁") base += 25;

      // ランダムで少しブレさせる
      let score = Math.min(100, base + Math.floor(Math.random() * 21) - 10);
      if (score < 0) score = 0;

      scoreEl.textContent = `今村優佑LOVE度 ${score}%`;

      if (score < 40) {
        messageEl.textContent = "まだまだ初心者。もっと優佑を知ろう！";
      } else if (score < 70) {
        messageEl.textContent = "なかなかの理解者！良きパートナー候補！";
      } else if (score < 90) {
        messageEl.textContent = "かなり愛が深い！もうすぐ100％！";
      } else {
        messageEl.textContent = "魂レベルでリンク済み。真のLOVE認定！💖";
      }

      form.classList.add("hidden");
      result.classList.remove("hidden");
    });
  </script>
